## Quickstart

```jsx
import React from "react";

import { IPhoneLayout } from "react-iphone-layout";

function App() {
  return (
    <IPhoneLayout>
      <p>this is iPhone layout!!!</p>
    </IPhoneLayout>
  );
}
```
